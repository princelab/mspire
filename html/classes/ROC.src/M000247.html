  def numhits_and_ppv(doublets)
    x = []
    y = []
    tps = 0
    fps = 0
    doublets.each_with_index do |d,i|
      if d[1] ; tps += 1
      else ; fps += 1 end

      if (i+1 == doublets.size) || (d[0] != doublets[i+1][0])
        num_hits = tps + fps
        x << num_hits
        y <<  tps.to_f/num_hits
      end
    end
    [x, y]
  end
