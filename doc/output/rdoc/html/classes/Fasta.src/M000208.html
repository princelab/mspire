  def initialize(arg=nil, filename=nil)
    @filename = filename
    @prots = []
    if arg 
      if arg.is_a? Fasta
        self.prots = arg.prots
        self.filename = arg.filename
      elsif arg.is_a? Array
        @prots = arg
      else
        read_file(arg)
      end
    end
    super(@prots)
  end
