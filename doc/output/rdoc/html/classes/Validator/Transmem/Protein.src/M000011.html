  def create_transmem_status_hash(peps)
    thash = {}
    peps.each do |pep|
      pep.prots.each do |prot|
        if !thash.key?(prot)
          #prot.transmem == nil
          thash[prot] = @transmem_by_ti_key[@transmem_index.reference_to_key(prot.reference)]
        end
      end
      if @no_include_tm_peps
        thash[pep] = pep_is_transmem?(pep)   
      end
    end
    thash
  end
